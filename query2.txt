Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia
SELECT s.*
FROM students s
JOIN degrees d ON s.degree_id = d.id
WHERE d.name = 'Corso di Laurea in Economia';

Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di Neuroscienze
SELECT d.*
FROM degrees d
JOIN departments dep ON d.department_id = dep.id
WHERE d.level = 'magistrale'
AND dep.name = 'Dipartimento di Neuroscienze';

Selezionare tutti i corsi in cui insegna Fulvio Amato (id=44)
SELECT c.*
FROM courses c
JOIN course_teacher ct ON c.id = ct.course_id
WHERE ct.teacher_id = 44;

Tutti gli studenti con corso di laurea e dipartimento, ordinati per cognome e nome
SELECT 
    s.name,
    s.surname,
    d.name AS corso_di_laurea,
    dep.name AS dipartimento
FROM students s
JOIN degrees d ON s.degree_id = d.id
JOIN departments dep ON d.department_id = dep.id
ORDER BY s.surname, s.name;

Tutti i corsi di laurea con relativi corsi e insegnanti
SELECT 
    d.name AS corso_di_laurea,
    c.name AS corso,
    t.name AS nome_docente,
    t.surname AS cognome_docente
FROM degrees d
JOIN courses c ON d.id = c.degree_id
JOIN course_teacher ct ON c.id = ct.course_id
JOIN teachers t ON ct.teacher_id = t.id;

Tutti i docenti che insegnano nel Dipartimento di Matematica
SELECT DISTINCT t.*
FROM teachers t
JOIN course_teacher ct ON t.id = ct.teacher_id
JOIN courses c ON ct.course_id = c.id
JOIN degrees d ON c.degree_id = d.id
JOIN departments dep ON d.department_id = dep.id
WHERE dep.name = 'Dipartimento di Matematica';

Numero tentativi per ogni esame e voto massimo >= 18
SELECT 
    s.id,
    s.name,
    s.surname,
    es.exam_id,
    COUNT(*) AS tentativi,
    MAX(es.vote) AS voto_massimo
FROM students s
JOIN exam_student es ON s.id = es.student_id
GROUP BY s.id, s.name, s.surname, es.exam_id
HAVING MAX(es.vote) >= 18;
